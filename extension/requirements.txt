# ---- required: Python indexing -------------------------------------------
# Jedi is why resolution works at all (85.7% on real code, zero false links) and
# it is Python-only. This is the one hard dependency.
jedi>=0.19

# ---- required: SQL (tables/columns, incl. SQL embedded in Python strings) --
sqlglot>=25.0

# ---- OPTIONAL: R only ------------------------------------------------------
# Nothing but the R front-end uses tree-sitter, and it is loaded lazily — a
# Python/SQL project indexes fine without this installed.
#
# This replaces tree_sitter_languages, which was abandoned: its last release
# predates Python 3.12, so it could not be installed on any current interpreter
# and R indexing was dead in practice. tree_sitter_language_pack is the
# maintained successor and works on 3.12+.
#
# It ships a NEWER r grammar with renamed nodes (left_assignment ->
# binary_operator, and friends). That rename fails SILENTLY — no error, no
# matches, an empty R index — so the node names live behind named adapters in
# spike/r_grammar.py, and test_render.py indexes fixtures/r_demo on every run to
# make sure the grammar and the code still agree.
tree_sitter_language_pack>=1.12
